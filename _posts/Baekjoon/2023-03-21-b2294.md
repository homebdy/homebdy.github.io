```python
import sys
input = sys.stdin.readline

n, k = map(int, input().split())
coins = []
dp = [0] * (k+1)

for _ in range(n):
    coins.append(int(input()))

for i in range(1, k+1):
    comparison = []
    for coin in coins:
        if i >= coin and dp[i - coin] != -1:
            comparison.append(dp[i - coin])
        if not comparison:
            dp[i] = -1
        else:
            dp[i] = min(comparison) + 1

print(dp[-1])
```
