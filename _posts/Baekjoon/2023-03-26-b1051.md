n, m = map(int, input().split())
numbers = [list(map(int, list(input()))) for _ in range(n)]
result = 1
size = min(n, m)

for s in range(0, size):
    for i in range(n-s):
        for j in range(m-s):
            if numbers[i][j] == numbers[i][j+s] == numbers[i+s][j] == numbers[i+s][j+s]:
                result = (s + 1) ** 2

print(result)
