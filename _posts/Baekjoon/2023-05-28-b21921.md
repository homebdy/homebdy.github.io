n, x = map(int, input().split())
visit = list(map(int, input().split()))

if max(visit) == 0:
    print("SAD")
else:
    max_visit = sum(visit[0:x])
    value = max_visit
    count = 1
    for i in range(x, n):
        value -= visit[i-x]
        value += visit[i]
        if value > max_visit:
            max_visit = value
            count = 1
        elif value == max_visit:
            count += 1
    print(max_visit)
    print(count)
